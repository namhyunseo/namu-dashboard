🕸️ LangGraph (상태 및 루프 관리자)

AI 에이전트의 비동기적 흐름을 '그래프(Graph)'로 모델링합니다. 공유 상태(State) 데이터를 관리하며, 검증 실패 시 특정 노드(Self-Correction)로 다시 분기하는 순환 루프 제어 기능 및 재귀 한계 차단(Recursion Limit) 기능을 담당합니다.

🛡️ TravelHarness Linter (결정론적 린터 노드)

LLM을 완전히 배제하고 100% 순수 파이썬 코드로 구성된 연산 기하 노드입니다. PostgreSQL 공간 DB(PostGIS)를 조회해 실제 도로 기반 거리를 쿼리하고, 정밀 예산을 가감 연산하여 위반 사항이 담긴 에러 리스트를 랭그래프 상태에 적어줍니다.

백엔드 엔진에 즉시 탑재하여 사용할 수 있는 LangGraph 정의 및 하네스 분기 제어 보일러플레이트 코드입니다. 

from typing import TypedDict, Annotated, List
from langgraph.graph import StateGraph, END

# 1. 그래프가 공유할 상태 객체 정의
class AgentState(TypedDict):
    itinerary_draft: dict
    harness_valid: bool
    harness_errors: List[str]
    loop_count: int

# 2. 하네스 검증기 노드 (LLM 없이 작동하는 순수 Python)
def travel_harness_node(state: AgentState):
    draft = state["itinerary_draft"]
    errors = []
    
    # PostGIS 공간 동선 Linter 실행
    distance_km = calculate_postgis_distance(draft)
    if distance_km > 50:
        errors.append(f"[ERROR] 동선 {distance_km}km: 일일 허용한도(50km) 초과")
        
    # 예산 한도 산술 검증
    total_cost = sum(draft.get("costs", []))
    if total_cost > 100000:
        errors.append(f"[ERROR] 총예산 {total_cost}원: 서연의 일정한도 초과")

    return {
        "harness_valid": len(errors) == 0,
        "harness_errors": errors,
        "loop_count": state["loop_count"] + 1
    }

# 3. 하네스 조건에 따른 분기용 라우터 함수
def harness_router(state: AgentState):
    if state["harness_valid"]:
        return "confirmed"  # 성공 시 완료 노드로 즉시 이관
    
    if state["loop_count"] >= 3:
        return "fallback"   # 3회 초과 에러 시 강제 보정용 노드로 이관
        
    return "self_correct"   # 자가 교정 노드로 회귀

# 4. LangGraph 그래프 빌딩 및 컴파일
workflow = StateGraph(AgentState)

workflow.add_node("draft_creator", draft_creator_node)
workflow.add_node("travel_harness", travel_harness_node)
workflow.add_node("self_correction", self_correction_node)
workflow.add_node("fallback_solver", fallback_solver_node)

workflow.set_entry_point("draft_creator")
workflow.add_edge("draft_creator", "travel_harness")
workflow.add_edge("self_correction", "travel_harness")

# 하네스 라우터 조건부 엣지 연동
workflow.add_conditional_edges(
    "travel_harness",
    harness_router,
    {
        "confirmed": END,
        "self_correct": "self_correction",
        "fallback": "fallback_solver"
    }
)
workflow.add_edge("fallback_solver", END)

app = workflow.compile()
